import collections

def compute_routing_paths(topology, requirements=None):
    # Build adjacency list
    lan_to_devices = collections.defaultdict(set)
    for device, ports in topology.items():
        for port, lan in ports.items():
            lan_to_devices[lan].add(device)
    
    graph = collections.defaultdict(list)
    for lan, devices in lan_to_devices.items():
        devices_list = list(devices)
        for i in range(len(devices_list)):
            for j in range(i+1, len(devices_list)):
                d1 = devices_list[i]
                d2 = devices_list[j]
                graph[d1].append(d2)
                graph[d2].append(d1)
    
    # Identify hosts
    hosts = [d for d in topology if d.startswith('h')]  # assumption: host names start with 'h'
    
    # Build result
    result = {}
    for src_host in hosts:
        result[src_host] = {}
        # BFS to all other hosts
        targets = {h for h in hosts if h != src_host}
        if not targets:
            continue
        # Use BFS
        visited = {src_host}
        queue = collections.deque([src_host])
        parent = {src_host: None}
        found = {}
        while queue and len(found) < len(targets):
            node = queue.popleft()
            if node in targets and node not in found:
                # reconstruct path
                path = []
                cur = node
                while cur is not None:
                    path.append(cur)
                    cur = parent[cur]
                path.reverse()
                found[node] = path
            for neighbor in graph.get(node, []):
                if neighbor not in visited:
                    visited.add(neighbor)
                    parent[neighbor] = node
                    queue.append(neighbor)
        for dst_host, path in found.items():
            result[src_host][dst_host] = path
    return result
